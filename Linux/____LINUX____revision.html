--Changer les autorisations d'un repertoire / dossier 
 
4 2 1 [DECIMAL]
1 0 0 [BINAIRE]
0 1 0 [BINAIRE]
0 0 1 [BINAIRE]
r w x [read(r), write(w), execute(x)

par exemple 700 = 111 000 000 = rwx --- --- pour le dossier Marie_secret_file 
pour autoriser que Marie a lire ecrire et executer 
dans le fichier]                    
        [root@fedora Marie]# chmod 700 /home/Marie/Marie_secret_file

par exemple 777 = 111 111 111 = rwx rwx rwx pour le dossier Public_file 
pour autoriser que marie a lire ecrire et executer 
dans le fichier                      
        [root@fedora Marie]# chmod 777 /home/Marie/Public_file

        [root@fedora Marie]# ll
        total 0
        drwx------ 1 Marie Marie 0  3 mar 19:38 Marie_secret_file

--The SetUID bit enforces user ownership 
on an executable file. When it is set, the file willexecute 
with the file owner's user ID, not the personrunning it.

ici on ajoute le SetUID au dossier Marie_secret_file            
        [root@fedora Marie]# chmod u+s /home/Marie/Marie_secret_file

--Ajoute l'autorisation spéciale STICKY-BIT qui protège les répertoires et leur contenu 
contre le piratage par des non-propriétaires.

d'une manière spéciale afin que seul lepropriétaire dufichier, 
le propriétaire du répertoireou l'utilisateur rootpuisse 
renommer ou supprimer lefichier

ici on ajout le STICKY-BIT au dossier Public_file

--Creer un dossier windows dans /mnt/FAT32 (mkdir /mnt/FAT32/)         
        [root@fedora ~]# mkdir /mnt/FAT32/windows

--La commande touch est principalement utilisée pour 
modifier les horodatages des fichiers, mais si le fichier (dont le nom est passé en argument) 
n'existe pas, alors l'outil le crée.

--ici on cree un fichier test.txt car il n'existe pas...
        [root@fedora ~]# touch /mnt/FAT32/windows/test.txt
